	\usepackage[american]{babel}%
	%\newcommand{\stripzero}[1]{\if0#1 0\else#1\fi}%
	\newcommand{\mymonthname}[1]{%
		\ifcase\stripzeros{#1} %
		\or Jan%
		\or Feb%
		\or Mar%
		\or Apr%
		\or May%
		\or Jun%
		\or Jul%
		\or Aug%
		\or Sep%
		\or Oct%
		\or Nov%
		\or Dec%
		\fi
	}
	\makeatletter
	\def\split@slash#1/#2/#3\@nil{\def\myyear{#1}\def\mymonth{#2}}
	\makeatother
	\newcommand{\thedate}[1]{%
		\setdate{#1}%
		\mymonthname\mymonth, \myyear%
	}
	\newcommand{\dateprefix}{{\tiny{from }}}%
	\newcommand{\datedelimiter}{{\tiny{to }}}%
	\newcommand{\datenow}{Now}
	\AtBeginDocument{\settowidth{\hintscolumnwidth}{{\dateprefix}Dec, 9999}}

			1 year%
		\else
			#1 years%
		\fi
	}
}
